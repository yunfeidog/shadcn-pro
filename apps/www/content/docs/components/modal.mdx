## Props

### Modal

| Prop           | Type                                     | Default   | Description                       |
| -------------- | ---------------------------------------- | --------- | --------------------------------- |
| `open`         | `boolean`                                | `-`       | Controls the modal open state     |
| `onOpenChange` | `(open: boolean) => void`                | `-`       | Callback when modal state changes |
| `trigger`      | `React.ReactNode`                        | `-`       | Element that triggers the modal   |
| `title`        | `string`                                 | `-`       | Modal title text                  |
| `description`  | `string`                                 | `-`       | Modal description text            |
| `variant`      | `default \| success \| error \| warning` | `default` | Modal variant style               |
| `children`     | `React.ReactNode`                        | `-`       | Modal content                     |
| `className`    | `string`                                 | `-`       | Additional CSS classes            |

## Advanced Usage

### Confirmation Dialog

```tsx showLineNumbers
import { useState } from "react"
import { Trash2 } from "lucide-react"

import { Modal } from "@/components/ui/modal"

function DeleteConfirmation() {
  const [open, setOpen] = useState(false)
  const [isDeleting, setIsDeleting] = useState(false)

  const handleDelete = async () => {
    setIsDeleting(true)
    try {
      // Perform delete operation
      await deleteItem()
      setOpen(false)
    } catch (error) {
      console.error("Delete failed:", error)
    } finally {
      setIsDeleting(false)
    }
  }

  return (
    <Modal
      variant="error"
      open={open}
      onOpenChange={setOpen}
      trigger={
        <button className="flex items-center gap-2 rounded-md bg-red-600 px-3 py-1 text-white hover:bg-red-700">
          <Trash2 className="h-4 w-4" />
          Delete Item
        </button>
      }
      title="Delete Item"
      description="This action cannot be undone. The item will be permanently deleted."
    >
      <div className="space-y-4">
        <p className="text-muted-foreground text-sm">
          Are you sure you want to delete this item? All associated data will be
          lost.
        </p>
        <div className="flex gap-2">
          <button
            onClick={() => setOpen(false)}
            disabled={isDeleting}
            className="bg-secondary text-secondary-foreground hover:bg-secondary/80 rounded-md px-4 py-2 disabled:opacity-50"
          >
            Cancel
          </button>
          <button
            onClick={handleDelete}
            disabled={isDeleting}
            className="rounded-md bg-red-600 px-4 py-2 text-white hover:bg-red-700 disabled:opacity-50"
          >
            {isDeleting ? "Deleting..." : "Delete"}
          </button>
        </div>
      </div>
    </Modal>
  )
}
```

### Form Modal

```tsx showLineNumbers
import { useState } from "react"

import { Modal } from "@/components/ui/modal"

function FormModal() {
  const [open, setOpen] = useState(false)
  const [formData, setFormData] = useState({ name: "", email: "" })

  const handleSubmit = (e: React.FormEvent) => {
    e.preventDefault()
    // Handle form submission
    console.log("Form data:", formData)
    setOpen(false)
  }

  return (
    <Modal
      open={open}
      onOpenChange={setOpen}
      trigger={<button>Add User</button>}
      title="Add New User"
      description="Enter the user details below"
    >
      <form onSubmit={handleSubmit} className="space-y-4">
        <div>
          <label htmlFor="name" className="block text-sm font-medium">
            Name
          </label>
          <input
            id="name"
            type="text"
            value={formData.name}
            onChange={(e) => setFormData({ ...formData, name: e.target.value })}
            className="mt-1 w-full rounded-md border px-3 py-2"
            required
          />
        </div>
        <div>
          <label htmlFor="email" className="block text-sm font-medium">
            Email
          </label>
          <input
            id="email"
            type="email"
            value={formData.email}
            onChange={(e) =>
              setFormData({ ...formData, email: e.target.value })
            }
            className="mt-1 w-full rounded-md border px-3 py-2"
            required
          />
        </div>
        <div className="flex gap-2 pt-4">
          <button
            type="button"
            onClick={() => setOpen(false)}
            className="bg-secondary text-secondary-foreground hover:bg-secondary/80 rounded-md px-4 py-2"
          >
            Cancel
          </button>
          <button
            type="submit"
            className="bg-primary text-primary-foreground hover:bg-primary/90 rounded-md px-4 py-2"
          >
            Add User
          </button>
        </div>
      </form>
    </Modal>
  )
}
```
